# JSON Design Tokens Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add conservative JSON/JSONC design token color highlighting with a configurable broad string mode.

**Architecture:** Add a pure `findJsonDesignTokens` strategy that scans JSON/JSONC string tokens, resolves full-string color values through existing detectors, and returns content-only ranges. Add one generated config setting and pass it through the existing highlight-run context. Update the registry so JSON/JSONC direct string colors are controlled by the JSON-aware strategy rather than the generic always-on strategies.

**Tech Stack:** TypeScript, reactive-vscode generated config, Vitest, playground file snapshots, `pnpm`, `tsdown`, `vscode-ext-gen`.

Create `src/strategies/json-design-tokens.ts`:

```ts
import { findColorFunctions } from './color-functions'
import { findHexRGBA } from './hex'
import { findHwb } from './hwb'
import { findNamedColors } from './named-colors'
import type { ColorDetector, ColorMatch, DesignTokenJsonMode } from '../types'

interface JsonStringToken {
  readonly start: number
  readonly end: number
  readonly contentStart: number
  readonly contentEnd: number
  readonly value: string
  readonly isPropertyKey: boolean
}

const TOKEN_VALUE_KEYS = new Set(['value', '$value'])

const VALUE_DETECTORS: readonly ColorDetector[] = [
  findHexRGBA,
  findColorFunctions,
  findHwb,
  findNamedColors,
]

export function findJsonDesignTokens(
  text: string,
  context?: { designTokenJsonMode?: DesignTokenJsonMode },
): ColorMatch[] {
  const mode = context?.designTokenJsonMode ?? 'token-values'
  if (mode === 'off') {
    return []
  }

  const matches: ColorMatch[] = []
  let pendingKey: string | undefined

  for (const token of scanJsonStrings(text)) {
    if (token.isPropertyKey) {
      pendingKey = token.value
      continue
    }

    const isTokenValue =
      pendingKey !== undefined && TOKEN_VALUE_KEYS.has(pendingKey)
    pendingKey = undefined

    const shouldMatch =
      mode === 'all' ||
      (mode === 'token-values' && isTokenValue) ||
      (mode === 'strings' && !isTokenValue)

    if (!shouldMatch) {
      continue
    }

    const color = resolveFullStringColor(token.value)
    if (!color) {
      continue
    }

    matches.push({
      start: token.contentStart,
      end: token.contentEnd,
      color,
    })
  }

  return dedupeMatches(matches)
}

function scanJsonStrings(text: string): JsonStringToken[] {
  const tokens: JsonStringToken[] = []
  let index = 0

  while (index < text.length) {
    const char = text[index]

    if (char === '/' && text[index + 1] === '/') {
      index = skipLineComment(text, index + 2)
      continue
    }

    if (char === '/' && text[index + 1] === '*') {
      index = skipBlockComment(text, index + 2)
      continue
    }

    if (char !== '"') {
      index++
      continue
    }

    const token = readJsonString(text, index)
    if (!token) {
      index++
      continue
    }

    tokens.push(token)
    index = token.end
  }

  return tokens
}

function readJsonString(text: string, start: number): JsonStringToken | null {
  let index = start + 1
  let value = ''

  while (index < text.length) {
    const char = text[index]

    if (char === '"') {
      const end = index + 1
      return {
        start,
        end,
        contentStart: start + 1,
        contentEnd: index,
        value,
        isPropertyKey: isStringFollowedByColon(text, end),
      }
    }

    if (char === '\\') {
      const escape = readEscape(text, index)
      value += escape.value
      index = escape.end
      continue
    }

    value += char
    index++
  }

  return null
}

function readEscape(
  text: string,
  start: number,
): { value: string; end: number } {
  const marker = text[start + 1]
  if (marker === undefined) {
    return { value: '\\', end: start + 1 }
  }

  if (marker === 'u') {
    const hex = text.slice(start + 2, start + 6)
    if (/^[\da-f]{4}$/iu.test(hex)) {
      return {
        value: String.fromCharCode(Number.parseInt(hex, 16)),
        end: start + 6,
      }
    }
  }

  const escapes: Record<string, string> = {
    '"': '"',
    '\\': '\\',
    '/': '/',
    b: '\b',
    f: '\f',
    n: '\n',
    r: '\r',
    t: '\t',
  }

  return {
    value: escapes[marker] ?? marker,
    end: start + 2,
  }
}

function isStringFollowedByColon(text: string, start: number): boolean {
  let index = start
  while (/\s/u.test(text[index] ?? '')) {
    index++
  }
  return text[index] === ':'
}

function skipLineComment(text: string, start: number): number {
  const nextLine = text.indexOf('\n', start)
  return nextLine === -1 ? text.length : nextLine + 1
}

function skipBlockComment(text: string, start: number): number {
  const end = text.indexOf('*/', start)
  return end === -1 ? text.length : end + 2
}

function resolveFullStringColor(value: string): string | null {
  for (const detector of VALUE_DETECTORS) {
    const matches = detector(value, {
      languageId: 'json',
      namedColorMatchMode: 'always',
    })
    if (matches instanceof Promise) {
      continue
    }

    const exact = matches.find(
      match => match.start === 0 && match.end === value.length,
    )
    if (exact) {
      return exact.color
    }
  }

  return null
}

function dedupeMatches(matches: ColorMatch[]): ColorMatch[] {
  const seen = new Set<string>()
  return matches.filter(match => {
    const key = `${match.start}:${match.end}:${match.color}`
    if (seen.has(key)) {
      return false
    }
    seen.add(key)
    return true
  })
}
```
